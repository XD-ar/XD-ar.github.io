### 代码块
``` cpp
class Solution {
public:
    int findRepeatNumber(vector<int>& nums) {
        int n=nums.size();
        for(int i=0;i<n;i++){
            if(nums[i]==i)
                continue;
            else{
                int k=nums[i];
                if(nums[k]==k){
                    return k;
                }
                else{
                    swap(nums[i],nums[k]);
                }
            }
        }
        return 0;
    }
};
```
